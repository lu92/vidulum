### 6.2 Update Handler Logic

```java
@CommandHandler
public UpdateRuleResult handle(UpdateRecurringRuleCommand cmd) {
    RecurringRule rule = ruleRepository.findById(cmd.ruleId())
        .orElseThrow(() -> new RuleNotFoundException(cmd.ruleId()));

    if (rule.getStatus() == RuleStatus.ENDED) {
        throw new RuleAlreadyEndedException(cmd.ruleId());
    }

    // Detect what changed
    RuleChangeSet changes = detectChanges(rule, cmd);

    if (changes.isEmpty()) {
        return UpdateRuleResult.noChanges(cmd.ruleId());
    }

    // 1. Update rule itself
    rule.update(cmd);
    ruleRepository.save(rule);

    // 2. Handle different change categories
    int transactionsUpdated = 0;
    int transactionsDeleted = 0;
    int transactionsCreated = 0;
    List<CashChangeId> affectedIds = new ArrayList<>();

    // Category A: Simple updates (name, description, notes, metadata)
    if (changes.hasSimpleUpdates()) {
        BatchUpdateResponse response = cashFlowClient.updateCashChangesBatch(
            rule.getCashFlowId(),
            rule.getId(),
            changes.getSimpleUpdates(),
            UpdateScope.ALL_EXPECTED
        );
        transactionsUpdated += response.count();
        affectedIds.addAll(response.updatedIds());
    }

    // Category B: Value updates (amount, category) - respects updateMode
    if (changes.hasValueUpdates()) {
        UpdateScope scope = cmd.updateMode() == UpdateMode.FUTURE_ONLY
            ? UpdateScope.FUTURE_EXPECTED
            : UpdateScope.ALL_EXPECTED;

        BatchUpdateResponse response = cashFlowClient.updateCashChangesBatch(
            rule.getCashFlowId(),
            rule.getId(),
            changes.getValueUpdates(),
            scope
        );
        transactionsUpdated += response.count();
        affectedIds.addAll(response.updatedIds());
    }

    // Category C: Schedule changes - delete and regenerate
    if (changes.hasScheduleChanges()) {
        // Delete future expected
        BatchDeleteResponse deleteResponse = cashFlowClient.deleteExpectedCashChanges(
            rule.getCashFlowId(),
            rule.getId(),
            LocalDate.now()  // from today onwards
        );
        transactionsDeleted = deleteResponse.deletedCount();

        // Regenerate with new schedule
        CashFlowInfo info = cashFlowClient.getCashFlowInfo(rule.getCashFlowId());
        List<ExpectedCashChangeItem> items = generationService.generateItems(
            rule, YearMonth.now(), info.horizon()
        );

        BatchCreateResponse createResponse = cashFlowClient.createExpectedCashChangesBatch(
            rule.getCashFlowId(),
            rule.getId(),
            items
        );
        transactionsCreated = createResponse.count();
        affectedIds.addAll(createResponse.createdIds());
    }

    // Category D: End condition changes
    if (changes.hasEndConditionChanges()) {
        handleEndConditionChange(rule, changes, affectedIds);
    }

    // Emit event
    eventEmitter.emit(new RecurringRuleUpdatedEvent(
        rule.getId(),
        rule.getCashFlowId(),
        cmd.updateMode(),
        changes.getUpdatedFields(),
        transactionsUpdated,
        transactionsDeleted,
        transactionsCreated,
        ZonedDateTime.now()
    ));

    return new UpdateRuleResult(
        cmd.ruleId(),
        changes.getUpdatedFields(),
        transactionsUpdated,
        transactionsDeleted,
        transactionsCreated,
        affectedIds
    );
}
```

```java
@CommandHandler
public DeleteRuleResult handle(DeleteRecurringRuleCommand cmd) {
    RecurringRule rule = ruleRepository.findById(cmd.ruleId())
        .orElseThrow(() -> new RuleNotFoundException(cmd.ruleId()));

    // Count transactions
    TransactionCounts counts = cashFlowClient.countTransactions(
        rule.getCashFlowId(),
        rule.getId()
    );

    int transactionsDeleted = 0;
    int transactionsKept = 0;

    if (cmd.deleteTransactions()) {
        // Delete only EXPECTED (never CONFIRMED)
        BatchDeleteResponse response = cashFlowClient.deleteExpectedCashChanges(
            rule.getCashFlowId(),
            rule.getId(),
            null  // all dates
        );
        transactionsDeleted = response.deletedCount();
        transactionsKept = counts.confirmed();
    } else {
        // Keep all - just unlink
        cashFlowClient.unlinkTransactionsFromRule(
            rule.getCashFlowId(),
            rule.getId()
        );
        transactionsKept = counts.expected() + counts.confirmed();
    }

    // Soft delete for undo
    String undoToken = UUID.randomUUID().toString();
    rule.softDelete(undoToken, cmd.deleteTransactions());
    ruleRepository.save(rule);

    // Schedule hard delete after undo window
    scheduler.schedule(
        () -> hardDeleteRule(rule.getId(), undoToken),
        Instant.now().plusSeconds(15)
    );

    // Emit event
    eventEmitter.emit(new RecurringRuleDeletedEvent(
        rule.getId(),
        rule.getCashFlowId(),
        cmd.deleteTransactions(),
        transactionsDeleted,
        ZonedDateTime.now()
    ));

    return new DeleteRuleResult(
        cmd.ruleId(),
        transactionsDeleted,
        transactionsKept,
        undoToken
    );
}
```

### 6.4 Undo Delete

```java
public record UndoDeleteRuleCommand(
    RecurringRuleId ruleId,
    String undoToken
) implements Command<Void> {}

@CommandHandler
public void handle(UndoDeleteRuleCommand cmd) {
    RecurringRule rule = ruleRepository.findById(cmd.ruleId())
        .orElseThrow(() -> new RuleNotFoundException(cmd.ruleId()));

    if (!rule.isDeleted()) {
        throw new RuleNotDeletedException(cmd.ruleId());
    }

    if (!rule.getUndoToken().equals(cmd.undoToken())) {
        throw new InvalidUndoTokenException();
    }

    if (rule.getDeletedAt().plusSeconds(15).isBefore(ZonedDateTime.now())) {
        throw new UndoWindowExpiredException();
    }

    // Restore rule
    rule.undoDelete();
    ruleRepository.save(rule);

    // If transactions were deleted, restore them too
    if (rule.getDeleteTransactionsOnDelete()) {
        cashFlowClient.restoreDeletedTransactions(
            rule.getCashFlowId(),
            rule.getId()
        );
    }

    // Emit event
    eventEmitter.emit(new RecurringRuleRestoredEvent(
        rule.getId(),
        rule.getCashFlowId(),
        ZonedDateTime.now()
    ));
}
```
